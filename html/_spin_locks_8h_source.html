502  {

504

506  const size_t owner = m_owner;

692  void CyclicSpinBarrier::wait() const
693  {
694  // Calls to wait while the counter is still resetting will block here
695  m_reset.lock(false);
696  assert(m_count > 0); // wait called too many times
697  const size_t count = m_count > 0 ? --m_count : 0;
698  {
699  if(count == 0)
700  {
701  /*
702  Relinquish our hold on the reentrant lock, grab one as a writer.
703  This will block until every other thread waiting has finished their wait
704  */
705  m_reset.unlock(false);
706  m_reset.lock(true);
707  }
708  while(m_count > 0)
709  {
710  // Spin out
711  }
712  if(count == 0)
713  {
714  m_count = m_initial;
715  m_reset.unlock(true);
716  }
717  }
718  // The "resetter" has already unlocked their hold on the reentrant lock
719  if(count != 0)
720  m_reset.unlock(false);
721  }
